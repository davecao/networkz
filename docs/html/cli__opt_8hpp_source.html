119  void ParseCmdLine(int argc, const char * argv[]) {
120  //
121  //parse command lines
122  po::variables_map vm;
123  try{
124  // initialize options
125  init();
126  po::store(parse_command_line(argc, argv, general),vm);
127  po::notify(vm);
128  }catch(po::ambiguous_option& e){
129  //std::cout<<"Program is terminated for the following reason(s):\n\n";
130  std::cout<<e.get_option_name()<<" is an ambiguous option name.\n";
131  std::cout << general << std::endl;
132  exit(0);
133  //BOOST_CHECK_EQUAL(std::string(e.what()), "Unknown option");
134  }catch(po::unknown_option& e){
135  std::cout<<e.get_option_name()<<" is an unknown option name.\n";
136  std::cout << general << std::endl;
137  exit(0);
138  //BOOST_CHECK_EQUAL(e.get_option_name(), "option name");
139  //BOOST_CHECK_EQUAL(std::string(e.what()), "Unknown option");
140  }
141 
142  if ( vm.count("help") ){
143  // print help info
144  std::cout << general << std::endl;
145  std::exit(0);
146  }
147  if ( vm.count("version") ){
148  //print version info
149  std::cout << VersionInfo::version << std::endl;
150  exit(0);
151  }
152  if ( vm.count("verbose") ){
153  verbose = true;
154  }
155  if ( vm.count("distance") ) {
156  // Check it was support or not?
157  auto found = std::find(d_type_supported.begin(),
158  d_type_supported.end(),
159  distance_type);
160  if (found == d_type_supported.end()) {
161  std::cerr << "Error: '-d, distance' option value INCORRECT."
162  << std::endl;
163  std::exit(-1);
164  }
165  }
166  if ( vm.count("infile") ) {
167  i_filename = vm["infile"].as<std::string>();
168  if( !FileExists(i_filename) ) {
169  std::cout << "Could not find the file: "<< i_filename <<std::endl;
170  std::cout << "Program terminated." << std::endl;
171  exit(0);
172  }
173  }else{
174  std::cout << general << std::endl;
175  std::exit(0);
176  }
177  }
